   ```
   const todayAnswers = unsorted.sort((a, b) => {

   ```

   ```
     if (aFollowed && !bFollowed) return -1
     if (!aFollowed && bFollowed) return 1
     // Then chronological
     return new Date(b.created_at) - new Date(a.created_at)
   })
   ```
